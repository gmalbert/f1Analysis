**Problem:** The `.apply()` with `wet_race_mask` filters before grouping, producing empty groups. The mask is applied to the global dataframe but used inside a per-group context, so every group is empty.

**File:** `f1-generate-analysis.py` — replace the broken block (line ~1801)

```python
# 4. Wet weather racecraft — FIXED: apply mask inside group, not globally
df['wet_race_vs_quali_delta'] = np.nan

wet_rows = df['total_precipitation'] > 0
if wet_rows.sum() > 20:
    # For each driver, compute the historical mean of (final_pos - qual_pos) in wet races
    wet_delta = (
        df.loc[wet_rows]
        .groupby('resultsDriverName')
        .apply(lambda g: (g['resultsFinalPositionNumber'] - g['resultsQualificationPositionNumber'])
               .shift(1).expanding().mean())
    )
    if hasattr(wet_delta, 'droplevel'):

    df['wet_race_vs_quali_delta'] = df.groupby('resultsDriverName')['wet_race_vs_quali_delta'].ffill()
    print(f"  wet_race_vs_quali_delta coverage: {df['wet_race_vs_quali_delta'].notna().mean():.1%}")
```

## 1C. New Tire Strategy Features (HIGH IMPACT)

The single biggest missing predictor category. F1 teams regard tire compound, stint count, and degradation as primary race-outcome drivers — yet these are completely absent from the current feature set.

### Step 1 — Data Pull Script: `f1-tire-strategy.py` (root directory)

```python
#!/usr/bin/env python3
"""Pull tire compound and stint data from FastF1 for all races 2018+.
Run once for historical backfill (~2-3 hours), then incrementally each week.
"""
import fastf1
import pandas as pd
import numpy as np
from pathlib import Path
import time
import os

os.environ['LOCAL_RUN'] = '1'  # Enable FastF1 cache

DATA_DIR = Path('data_files')
OUTPUT_FILE = DATA_DIR / 'tire_strategy_data.csv'
CACHE_DIR = DATA_DIR / 'f1_cache'
fastf1.Cache.enable_cache(str(CACHE_DIR))

existing = set()
if OUTPUT_FILE.exists():
    old = pd.read_csv(OUTPUT_FILE, sep='\t')
    existing = set(zip(old['year'], old['round']))

all_rows = []

for year in range(2018, 2027):
    try:
        schedule = fastf1.get_event_schedule(year, include_testing=False)
    except Exception as e:
        print(f"  Skip year {year}: {e}")
        continue

    for _, event in schedule.iterrows():
        rnd = int(event['RoundNumber'])
        if rnd == 0 or (year, rnd) in existing:
            continue
        try:
            session = fastf1.get_session(year, rnd, 'R')
            session.load(laps=True, telemetry=False, weather=False, messages=False)
        except Exception as e:
            print(f"  Skip {year} R{rnd}: {e}")
            continue

        laps = session.laps
        if laps is None or laps.empty:
            continue

        for driver in laps['Driver'].unique():
            dl = laps[laps['Driver'] == driver].sort_values('LapNumber')
            if dl.empty:
                continue

            stints = dl.groupby('Stint').agg(
                compound=('Compound', 'first'),
                laps_in_stint=('LapNumber', 'count'),
                avg_lap_sec=('LapTime', lambda x: x.dt.total_seconds().mean()),
            ).reset_index()

            num_stints = len(stints)
            starting_compound = stints.iloc[0]['compound'] if num_stints > 0 else None

            # Tire degradation: pace diff between first 3 and last 3 laps per stint
            stint_degs = []
            for _, stint_row in stints.iterrows():
                sl = dl[dl['Stint'] == stint_row['Stint']]
                lap_secs = sl['LapTime'].dt.total_seconds().dropna()
                if len(lap_secs) >= 6:
                    stint_degs.append(lap_secs.iloc[-3:].mean() - lap_secs.iloc[:3].mean())
            avg_degradation = float(np.mean(stint_degs)) if stint_degs else np.nan

            total_laps = int(dl['LapNumber'].max()) if not dl.empty else 0
            soft_laps = int((dl['Compound'] == 'SOFT').sum())
            soft_ratio = soft_laps / total_laps if total_laps > 0 else 0

            all_rows.append({
                'year': year, 'round': rnd,
                'event_name': str(event.get('EventName', '')),
                'driver': str(driver),
                'num_stints': num_stints,
                'starting_compound': str(starting_compound),
                'avg_stint_length': float(stints['laps_in_stint'].mean()),
                'max_stint_length': int(stints['laps_in_stint'].max()),
                'avg_tire_degradation_sec': avg_degradation,
                'soft_ratio': soft_ratio,
                'used_hard': int('HARD' in stints['compound'].values),
                'used_medium': int('MEDIUM' in stints['compound'].values),
                'used_soft': int('SOFT' in stints['compound'].values),
                'total_laps': total_laps,
            })

        print(f"  Processed {year} R{rnd} ({event.get('EventName', '')})")
        time.sleep(0.3)

result = pd.DataFrame(all_rows)
if OUTPUT_FILE.exists():
    old = pd.read_csv(OUTPUT_FILE, sep='\t')
    result = pd.concat([old, result], ignore_index=True).drop_duplicates(
        subset=['year', 'round', 'driver'], keep='last')
result.to_csv(OUTPUT_FILE, sep='\t', index=False)
print(f"Saved {len(result)} rows to {OUTPUT_FILE}")
```

```python
        df.groupby('circuitId')['avg_tire_degradation_sec']

        df['driver_avg_tire_degradation'] * df['track_tire_degradation'])
```

## 1D. Race Lap Pace Features (HIGH IMPACT)

Fuel-corrected pace and sector consistency from actual race lap data fill a major gap — the current model has qualifying lap data but no equivalent race-lap pace features.

### Step 1 — Data Pull Script: `f1-race-pace-laps.py` (root directory)

```python
#!/usr/bin/env python3
"""Pull race lap-level pace data from FastF1.
Used for fuel-corrected pace, race sector consistency, and completion reliability.
"""
import fastf1
import pandas as pd
import numpy as np
from pathlib import Path
import time
import os

os.environ['LOCAL_RUN'] = '1'

DATA_DIR = Path('data_files')
OUTPUT_FILE = DATA_DIR / 'race_pace_lap_data.csv'
CACHE_DIR = DATA_DIR / 'f1_cache'
fastf1.Cache.enable_cache(str(CACHE_DIR))

existing = set()
if OUTPUT_FILE.exists():
    old = pd.read_csv(OUTPUT_FILE, sep='\t')
    existing = set(zip(old['year'], old['round']))

all_rows = []

for year in range(2018, 2027):
    try:
        schedule = fastf1.get_event_schedule(year, include_testing=False)
    except Exception as e:
        print(f"  Skip year {year}: {e}")
        continue

    for _, event in schedule.iterrows():
        rnd = int(event['RoundNumber'])
        if rnd == 0 or (year, rnd) in existing:
            continue
        try:
            session = fastf1.get_session(year, rnd, 'R')
            session.load(laps=True, telemetry=False, weather=False, messages=False)
        except Exception as e:
            print(f"  Skip {year} R{rnd}: {e}")
            continue

        laps = session.laps
        if laps is None or laps.empty:
            continue

        total_race_laps = int(laps['LapNumber'].max())

        for driver in laps['Driver'].unique():
            dl = laps[laps['Driver'] == driver].copy()
            dl['LapTime_sec'] = dl['LapTime'].dt.total_seconds()
            dl = dl[dl['LapTime_sec'].notna() & (dl['LapTime_sec'] > 0)]
            if len(dl) < 10:
                continue

            early_pace = dl.head(10)['LapTime_sec'].mean()
            late_pace = dl.tail(10)['LapTime_sec'].mean()
            fuel_corrected = (
                (late_pace - early_pace) / early_pace * 100
                if early_pace > 0 else np.nan
            )

            # Exclude pit laps for consistency metric
            regular = dl[dl['PitInTime'].isna()] if 'PitInTime' in dl.columns else dl
            pace_std = float(regular['LapTime_sec'].std()) if len(regular) > 5 else np.nan

            all_rows.append({
                'year': year, 'round': rnd,
                'driver': str(driver),
                'fuel_corrected_pace_pct': fuel_corrected,
                'race_pace_std': pace_std,
                'avg_lap_time_sec': float(dl['LapTime_sec'].mean()),
                'laps_completed': len(dl),
                'total_race_laps': total_race_laps,
                'completion_pct': len(dl) / total_race_laps if total_race_laps > 0 else 0,
            })

        print(f"  Processed {year} R{rnd}")
        time.sleep(0.3)

result = pd.DataFrame(all_rows)
if OUTPUT_FILE.exists():
    old = pd.read_csv(OUTPUT_FILE, sep='\t')
    result = pd.concat([old, result], ignore_index=True).drop_duplicates(
        subset=['year', 'round', 'driver'], keep='last')
result.to_csv(OUTPUT_FILE, sep='\t', index=False)
print(f"Saved {len(result)} rows to {OUTPUT_FILE}")
```

```python
# --- RACE LAP PACE FEATURES ---
race_pace_file = os.path.join(DATA_DIR, 'race_pace_lap_data.csv')
if os.path.exists(race_pace_file):
    race_pace = pd.read_csv(race_pace_file, sep='\t')
    df = df.merge(race_pace,
        left_on=['grandPrixYear', 'round', 'Abbreviation'],
        right_on=['year', 'round', 'driver'],
        how='left', suffixes=('', '_rp'))

    df['driver_fuel_corrected_pace'] = (
        df.groupby('resultsDriverName')['fuel_corrected_pace_pct']
        .transform(lambda x: x.shift(1).rolling(5, min_periods=1).mean()))

    df['driver_race_pace_std'] = (
        df.groupby('resultsDriverName')['race_pace_std']
        .transform(lambda x: x.shift(1).rolling(5, min_periods=1).mean()))

    df['track_race_pace_std'] = (
        df.groupby(['circuitId', 'resultsDriverName'])['race_pace_std']
        .transform(lambda x: x.shift(1).rolling(3, min_periods=1).mean()))

    # Race consistency relative to qualifying consistency
    df['race_vs_qual_consistency'] = (
        df['driver_race_pace_std'].fillna(0) /
        df['qualifying_consistency_std'].fillna(1).replace(0, 1))

    df['driver_avg_completion_pct'] = (
        df.groupby('resultsDriverName')['completion_pct']
        .transform(lambda x: x.shift(1).rolling(10, min_periods=1).mean()))

    print("Created 5 race lap pace features")
```

```python
# Tire management strength vs circuit corner count
```
